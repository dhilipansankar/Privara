### **STEP 2: BUILD WINDOWS .EXE**

#### **2.1 Prerequisites**

```powershell
# Check Java installation
java -version  # Should show Java 21+

# Check Maven installation
mvn -version   # Should show Maven 3.8+

# Install NSIS (Nullsoft Scriptable Install System)
# Download from: https://nsis.sourceforge.io/
# Add NSIS to PATH: C:\Program Files (x86)\NSIS
```

#### **2.2 Build the EXE**

```batch
# From project root directory
cd privara-hids

# Clean and build JAR
mvn clean package -DskipTests

# Build NSIS installer
cd installer\nsis
"C:\Program Files (x86)\NSIS\makensis.exe" /V4 Privara.nsi

# Output: Privara-Setup-1.0.0.exe
cd ..\..

# Generate SHA-256 hash for integrity verification
certutil -hashfile installer\nsis\Privara-Setup-1.0.0.exe SHA256 > RELEASE_HASH.txt
```

#### **2.3 Code Signing (Optional but Recommended)**

```batch
# Obtain EV Code Signing Certificate from DigiCert
# Export as PKCS#12 (.pfx)

# Sign the executable
signtool sign /f cert.pfx /p PASSWORD /d "Privara Enterprise HIDS" ^
  /du "https://github.com/privara-hids/privara-hids" ^
  /t "http://timestamp.digicert.com" ^
  installer\nsis\Privara-Setup-1.0.0.exe

# Verify signature
signtool verify /v /pa installer\nsis\Privara-Setup-1.0.0.exe
```

### **GitHub Actions Workflows**

**`.github/workflows/build.yml`** - Automated testing on push:

```yaml
name: Build

on: [push, pull_request]

jobs:
  build:
    runs-on: windows-latest
    strategy:
      matrix:
        java-version: ['21']
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Set up Java
      uses: actions/setup-java@v3
      with:
        java-version: ${{ matrix.java-version }}
        distribution: 'temurin'
    
    - name: Build with Maven
      run: mvn clean package -DskipTests
    
    - name: Upload JAR
      uses: actions/upload-artifact@v3
      with:
        name: Privara-${{ github.run_number }}
        path: target/Privara-1.0.0.jar
```

```powershell
# Download and install in one command
$url = "https://github.com/privara-hids/privara-hids/releases/download/v1.0.0/Privara-Setup-1.0.0.exe"

### **Admin Deployment Script**

```batch
@echo off
REM Deploy Privara to 100+ workstations via domain

set SHARE=\\deployment-server\software\privara
set VERSION=1.0.0

REM Create package
mkdir %SHARE%
copy Privara-Setup-%VERSION%.exe %SHARE%\
copy RELEASE_HASH.txt %SHARE%\

REM Deploy via Group Policy (SYSVOL)
copy %SHARE%\Privara-Setup-%VERSION%.exe %systemroot%\SYSVOL\sysvol\domain\Policies\

echo Deployment package ready for Group Policy distribution
```

### **Update Existing Installations**

```python
#!/usr/bin/env python3
# Update all Privara instances to latest version

import os
import subprocess
import requests
import json

def get_latest_release():
    """Fetch latest Privara version from GitHub API"""
    url = "https://api.github.com/repos/privara-hids/privara-hids/releases/latest"
    response = requests.get(url)
    return response.json()['tag_name'].lstrip('v')

def update_privara(current_version, latest_version):
    """Download and install latest version"""
    if current_version < latest_version:
        url = f"https://github.com/privara-hids/privara-hids/releases/download/v{latest_version}/Privara-Setup-{latest_version}.exe"
        
        # Download
        subprocess.run(f"powershell -Command \"Invoke-WebRequest -Uri {url} -OutFile $env:TEMP\\Privara-Update.exe\"")
        
        # Stop service
        subprocess.run("taskkill /IM Privara.exe /F")
        subprocess.run("net stop PrivaraService")
        
        # Install
        subprocess.run(f"$env:TEMP\\Privara-Update.exe /S")
        
        # Verify
        subprocess.run("net start PrivaraService")
        print(f"Updated Privara from {current_version} to {latest_version}")

if __name__ == "__main__":
